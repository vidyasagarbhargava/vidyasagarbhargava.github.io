cars = sns.load_dataset('mpg')

def bandwidth_widget(bw=0.20):
    sns.kdeplot(cars.horsepower, lw=3, fill=True, bw_adjust=bw)
    plt.xlim(-30, 300)
    plt.ylim(0, 0.03);
