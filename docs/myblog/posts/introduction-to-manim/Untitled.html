Introduction

A community maintained Python library for creating mathematical animations.


%%manim -qm SimpleExample

class SimpleExample(Scene):
    def construct(self):
        blue_circle = Circle(color='#654a4e', fill_opacity=0.5)
        green_square = Square(color='#20B2AA', fill_opacity=0.5)
        green_square.next_to(blue_circle, RIGHT)
        self.add(blue_circle, green_square)

%%manim -qm AnimationExample
class AnimationExample(Scene):
    def construct(self):

        curve = ax.plot(lambda x: (x+2)*x*(x-2)/2, color='#654a4e')

Getting the area

%%manim -qm AnimationExample
class AnimationExample(Scene):
    def construct(self):
        ax = Axes(x_range =(-3,3), y_range=(-3,3))
        curve = ax.plot(lambda x: (x+2)*x*(x-2)/2, color='#654a4e')
        area = ax.get_area(curve, x_range=(-2,0))
        self.add(ax,curve, area)

Now lets see some animation


%%manim -v WARNING  --progress_bar None   --disable_caching AnimationExample
config.media_embed = True


class AnimationExample(Scene):
    def construct(self):
        self.camera.background_color = '#e6ddde'
        ax = Axes(x_range =(-3,3), y_range=(-3,3))
        curve = ax.plot(lambda x: (x+2)*x*(x-2)/2, color='#654a4e')
        area = ax.get_area(curve, x_range=(-2,0))
        self.add(ax,curve, area)
        self.play(Create(ax), Create(curve))
        self.play(FadeIn(area))
